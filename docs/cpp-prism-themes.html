vec3.hpp — a small math type

#include <cmath>
#include <vector>
#include <random>

// Minimal 3D vector for a Monte Carlo path tracer
template <typename T>
struct Vec3 {
    T x, y, z;

    constexpr Vec3(T x = 0, T y = 0, T z = 0) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator*(T s)          const { return {x * s,   y * s,   z * s};   }

    T dot(const Vec3& o) const { return x * o.x + y * o.y + z * o.z; }
    T length()           const { return std::sqrt(dot(*this)); }
};

using Vec3f = Vec3<float>;
using Vec3d = Vec3<double>;

radiance.cpp — recursive path tracing

// Estimate incoming radiance along a ray by recursive path tracing.
Vec3f radiance(const Ray& ray, const Scene& scene, int depth, RngState& rng) {
    constexpr int kMaxDepth = 8;
    if (depth >= kMaxDepth)
        return Vec3f{0.0f, 0.0f, 0.0f};

    HitRecord hit;
    if (!scene.intersect(ray, hit))
        return scene.sky(ray.direction);          // miss: sample the environment

    // Russian roulette after a few bounces to keep paths unbiased.
    const float p = std::max({hit.albedo.x, hit.albedo.y, hit.albedo.z});
    if (depth > 3 && next_float(rng) > p)
        return hit.emission;

    const Ray  scattered  = sample_bsdf(hit, ray, rng);
    const Vec3f throughput = hit.albedo * (1.0f / std::max(p, 1e-4f));

    return hit.emission + throughput * radiance(scattered, scene, depth + 1, rng);
}

Inline code — themed vs. page-styled

Inline snippets such as std::vector<Vec3f> or throughput *= albedo carry the language-cpp class, so Prism tokenizes them and the selected theme paints their background — switch themes above and watch them change along with the blocks. By contrast, a plain chip like OnRenderImage() or g_Radiance has no language class: Prism leaves it alone, so it keeps the page’s own styling no matter which theme is active.
